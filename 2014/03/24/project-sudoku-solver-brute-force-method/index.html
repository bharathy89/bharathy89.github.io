boolean solveSudoku(int[][] sudokumatrix) { 
    boolean solved = false;
    for(int i = 0; i< 9; i++) {
        for(int j =0; j <9; j++) {
            if(sudokumatrix[i][j] == 0) {
                int k = 1;
                count++;
                for( k = 1; k < 10; k++) {
                    boolean alreadyPresent = false;

                    // check in the row.
                    for(int l= 0 ; l< 9; l++) {
                        if(sudokumatrix[i][l] == k && l != j){
                            alreadyPresent = true;
                            break;
                        }
                    }

                    // check in the column.
                    if(!alreadyPresent){
                        for(int l= 0; l < 9; l++) {
                            if(sudokumatrix[l][j] == k && l != i) {
                                alreadyPresent = true;
                                break;
                            }
                        }
                    }

                    // check in the box.
                    if(!alreadyPresent){
                        int m = 0, n= 0;
                        if(0 <= i && i < 3 ) {
                            m = 0;
                        }else if(3<=i && i < 6){
                            m = 3;
                        }else if(6<=i && i < 9){
                            m = 6;
                        }

                        if(0 <= j && j < 3 ) {
                            n = 0;
                        }else if(3<=j && j < 6){
                            n = 3;
                        }else if(6<=j && j < 9){
                            n = 6;
                        }

                        for(int m1 = m; m1 < m+3; m1++){
                            for(int n1 = n; n1<n+3; n1++){
                                if(m1!= i && n1 !=j && sudokumatrix[m1][n1]== k){
                                    alreadyPresent = true;
                                    break;
                                }
                            }
                            if(alreadyPresent)
                                break;
                        }
                    }
                    if(!alreadyPresent){
                        sudokumatrix[i][j] = k;
                        if(solveSudoku(sudokumatrix)) {
                        // k value solved the sudoku for that cell.
                            return true;
                        }else {
                            // reverting since this k value doesnt solve the sudoku.
                            sudokumatrix[i][j] = 0;
                            solved = false;
                        }
                    }
                }
                // If you reach this part it means there is no value that we can put in this cell.
                if (k == 10){
                    return false;
                }
            }   
        }
    }
    // If you reach this part that means there is no empty cell in the sudoku is.
    return true;
}
